# Hex

## Install
`npm i --save @dwlib/hex`

## Usage
```javascript
// CJS
const Hex = require('@dwlib/hex');
const HexDecode = require('@dwlib/hex/decode');
const HexDecodeBigInt = require('@dwlib/hex/decodeBigInt');
const HexDecodeInt = require('@dwlib/hex/decodeInt');
const HexDecodeInto = require('@dwlib/hex/decodeInto');
const HexEncode = require('@dwlib/hex/encode');
const HexEncodeBigInt = require('@dwlib/hex/encodeBigInt');
const HexEncodeInt = require('@dwlib/hex/encodeInt');
const HexIsValid = require('@dwlib/hex/isValid');
const HexNormalize = require('@dwlib/hex/normalize');
const HexValidate = require('@dwlib/hex/validate');
// ESM
import Hex from '@dwlib/hex';
import HexDecode from '@dwlib/hex/decode';
import HexDecodeBigInt from '@dwlib/hex/decodeBigInt';
import HexDecodeInt from '@dwlib/hex/decodeInt';
import HexDecodeInto from '@dwlib/hex/decodeInto';
import HexEncode from '@dwlib/hex/encode';
import HexEncodeBigInt from '@dwlib/hex/encodeBigInt';
import HexEncodeInt from '@dwlib/hex/encodeInt';
import HexIsValid from '@dwlib/hex/isValid';
import HexNormalize from '@dwlib/hex/normalize';
import HexValidate from '@dwlib/hex/validate';
```
